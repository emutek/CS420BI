int theidfunc(int i)
{
    float f;
f = i;
    return f;
}

int faccctorial(int n)
{
    if (n < 2) return 1;
    return n * faccctorial(n-1);
}

float basicrecfunc(float fafafa)
{
    if (fafafa < 1) return 1;
    return fafafa * basicrecfunc(fafafa - 1);
}
